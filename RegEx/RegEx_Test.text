    <script>
        function validation()
        {
            var uname = document.getElementById('uname').value;
            var pwd = document.getElementById('pwd').value;

            var unamecheck = /[A-Za-z. ]{3,30}/;
            var pwdcheck = /(?=.*[0-9])(?=.*[!@#$%^&*])[a-zA-z0-9!@#$%^&*]{8,16}/;


            if (unamecheck.test(uname)) {
                document.getElementById('uname_error').innerHTML = "Valid";

            }
            else {
                document.getElementById('uname_error').innerHTML = "Enter name using alphabets only";
                return false;
            }
            
            if (pwdcheck.test(pwd))
            {
                document.getElementById('pwd_error').innerHTML = "Valid";

            }
            else {
                document.getElementById('pwd_error').innerHTML = "Enter 8-cahracters password using numbers, alphabets & special character";
                return false;
            }
            
        }
    </script>

</head>
<body style="background-color:paleturquoise">

    <div class="container">
        <h2 style="color:darkblue">LOGIN TO CONTINUE</h2>
        
        <form action="#" onsubmit="return validation()" class="was-validated">
            <div class="form-group">
                <label for="uname">Username:</label>
                <input type="text" class="form-control" id="uname" placeholder="Enter username" name="pswd" required>
                <span id="uname_error" class="text-danger font-weight-bold"> </span>
                <div class="invalid-feedback">Please fill out this field.</div>
            </div>
            <div class="form-group">
                <label for="pwd">Password:</label>
                <input type="password" class="form-control" id="pwd" placeholder="Enter password" name="pswd" required>    
                <span id="pwd_error" class="text-danger font-weight-bold"> </span>
                <div class="invalid-feedback">Please fill out this field.</div>
               
            </div>
            <div class="form-group form-check">
                <label class="form-check-label">
                    <input class="form-check-input" type="checkbox" name="remember" required> I accept the terms and conditions 
                    <div class="valid-feedback">Valid.</div>
                    <div class="invalid-feedback">Check this checkbox to continue.</div>
                </label>
            </div>
            
            <input type="submit" name="submit" value="submit" id="submit" class="btn btn-primary">
        </form>
    </div>
